```python
"""Convert vendor tag columns to JSONB, add GIN indexes and sighting compound index.

Revision ID: (generate random hex)
Revises: (current head)
"""

import sqlalchemy as sa
from alembic import op


revision = "(random hex)"
down_revision = "(current head)"
branch_labels = None
depends_on = None


def _index_exists(index_name, table_name):
    from sqlalchemy import inspect
    bind = op.get_bind()
    insp = inspect(bind)
    return any(ix["name"] == index_name for ix in insp.get_indexes(table_name))


def upgrade():
    # Convert JSON → JSONB for containment queries and GIN indexing
    op.execute("ALTER TABLE vendor_cards ALTER COLUMN brand_tags TYPE JSONB USING brand_tags::jsonb")
    op.execute("ALTER TABLE vendor_cards ALTER COLUMN commodity_tags TYPE JSONB USING commodity_tags::jsonb")

    # GIN indexes for future filter support
    if not _index_exists("ix_vendor_brand_tags", "vendor_cards"):
        op.execute("CREATE INDEX ix_vendor_brand_tags ON vendor_cards USING gin(brand_tags jsonb_path_ops)")
    if not _index_exists("ix_vendor_commodity_tags", "vendor_cards"):
        op.execute("CREATE INDEX ix_vendor_commodity_tags ON vendor_cards USING gin(commodity_tags jsonb_path_ops)")

    # Compound index for MPN+vendor sighting lookup
    if not _index_exists("ix_sightings_mpn_vendor_norm", "sightings"):
        op.create_index(
            "ix_sightings_mpn_vendor_norm",
            "sightings",
            ["normalized_mpn", "vendor_name_normalized"],
        )


def downgrade():
    op.drop_index("ix_sightings_mpn_vendor_norm", table_name="sightings")
    op.execute("DROP INDEX IF EXISTS ix_vendor_commodity_tags")
    op.execute("DROP INDEX IF EXISTS ix_vendor_brand_tags")
    op.execute("ALTER TABLE vendor_cards ALTER COLUMN commodity_tags TYPE JSON USING commodity_tags::json")
    op.execute("ALTER TABLE vendor_cards ALTER COLUMN brand_tags TYPE JSON USING brand_tags::json")
```

```python
@router.get("/v2/partials/vendors/find-by-part", response_class=HTMLResponse)
async def find_by_part_partial(
    request: Request,
    mpn: str = "",
    hx_target: str = Query("#main-content", alias="hx_target"),
    push_url_base: str = Query("/v2/vendors", alias="push_url_base"),
    user: User = Depends(require_user),
    db: Session = Depends(get_db),
):
    """Search vendors by MPN via MaterialVendorHistory."""
    from app.utils.normalization import normalize_mpn

    hx_target, push_url_base = _sanitize_hx_params(hx_target, push_url_base, "/v2/vendors")

    results = []
    norm_mpn = normalize_mpn(mpn) if mpn.strip() else None

    if norm_mpn:
        rows = (
            db.query(MaterialVendorHistory, VendorCard)
            .join(MaterialCard, MaterialVendorHistory.material_card_id == MaterialCard.id)
            .outerjoin(VendorCard, VendorCard.normalized_name == MaterialVendorHistory.vendor_name_normalized)
            .filter(MaterialCard.normalized_mpn == norm_mpn)
            .order_by(
                MaterialVendorHistory.times_seen.desc(),
                VendorCard.response_rate.desc().nullslast(),
                VendorCard.total_pos.desc().nullslast(),
                VendorCard.avg_response_hours.asc().nullslast(),
            )
            .limit(30)
            .all()
        )
        for mvh, vc in rows:
            results.append({
                "vendor_name": mvh.vendor_name,
                "vendor_id": vc.id if vc else None,
                "times_seen": mvh.times_seen or 0,
                "last_price": mvh.last_price,
                "last_qty": mvh.last_qty,
                "last_seen": mvh.last_seen,
                "win_rate": vc.overall_win_rate if vc else None,
                "avg_response_hours": vc.avg_response_hours if vc else None,
            })

    ctx = _base_ctx(request, user, "vendors")
    ctx.update({
        "mpn": mpn.strip().upper() if mpn.strip() else None,
        "results": results,
        "hx_target": hx_target,
        "push_url_base": push_url_base,
    })
    return templates.TemplateResponse("htmx/partials/vendors/find_by_part.html", ctx)
```
